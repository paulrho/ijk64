085/** Machine modules :
086<PRE>
087    +-------------------------
088    | VStore : Variable Store
089    +-------------------------
090    | FStack : FOR/NEXT Stack
091    | GStack : Gosub Stack
092    +-------------------------
093    | PText  : Program/Direct text and exection/data points
094    |   ProgramText                  : The program BASIC code itself
095    |   DirectText                   : The line being executed in immediate (direct) mode [notionally exists]
096    |   ---------------------
097    |   executionpoint               : The point in the program we are actually parsing/executing
098    |                                  not always kept up to date (pnt in statements is up to date)
099    |   program_saved_executionpoint : Saved for use in CONT
100    |   uptoDATA                     : The point in the program we are upto with DATA READ
101    +-------------------------
102    | MemoryIO                       : Pokes/peeks/sys
103    +-------------------------
104    | LCache : Line Cache
105    | DCache : Data Cache
106    +-------------------------
107    | ScreenIO                       : Hooks to C64Screen (if exists)
108    +-------------------------
109</PRE>
110  <BR>
111<PRE>
112   Object interdependance :
113
114                  ,----+ C64Screen           (precreated)
115                 / ,+--+ C64PopupMenu        (precreated)               {+-+links back to machine}
116                / /
117   C64-+ Machine 
118                 -+ FStack                   {internal}
119                 -+ GStack                   {internal}
120                 -+ PText                    {internal, including execution/data/save points}
121                 -+ LCache                   {internal}
122                 -+ DCache                   {internal}
123           -+ Variables                      (Variable Store (VStore))
124           +-+ evaluate                      (evaluate engine)          [ could be renamed : ExpressionEvaluator ]
125                 interpret_string ( string )                            {+-+links back to machine}
126           +-+ statements ( string )         (parse the program/direct) [ could be renamed : CxxParser ]
127               [instansiated each time]                                 {+-+links back to machine}
128  
129   Passing datatype : GenericType
130   Exceptions       : BasicException and EvaluateException
131</PRE>
132  <BR>
133  Notes:
134  In immediate (direct) mode, we cannot push onto GStack, but we can push onto FStack
135  Pulling off either (by doing RETURN or NEXT) doesnt really make sense, but will do

346  boolean processNEXT(int current, String var) throws BasicException {
347    if (verbose) { System.out.printf("processing NEXT %s at current=%d\n",var,current); }
348    // could be multiple steps?, no, this should be dealt with by the statements parser
349    int fl=topforloopstack;
350    executionpoint=current;
351    while(fl>0) {
352      fl--;
353      if (forloopstack_var[fl].startsWith("_")) break; // matches c64 behavior 
354      if (var.equals("") || forloopstack_var[fl].equals(var)) {
355      //if (var.equals("") && !forloopstack_var[fl].startsWith("_") || forloopstack_var[fl].equals(var)) {
356        // just pop the last one
357        setvariable(forloopstack_var[fl].toLowerCase(),evaluate(forloopstack_var[fl].toLowerCase()+"+"+forloopstack_step[fl]));
358        if (verbose) { System.out.printf("about to add to loop at stack location %d %f>%f\n",fl,getvariable(forloopstack_var[fl]).num(),forloopstack_to[fl]); }
359        //if (verbose) { dumpstate(); }
360        if (forloopstack_step[fl]<0 && getvariable(forloopstack_var[fl].toLowerCase()).num()<forloopstack_to[fl]  
361            || forloopstack_step[fl]>0 && getvariable(forloopstack_var[fl].toLowerCase()).num()>forloopstack_to[fl]) {
362          if (verbose) { System.out.printf("NEXT: At end of stack (%s)\n",forloopstack_var[fl].toLowerCase()); }
363          // pop it off, but keep on going, we may be popping off many
364          topforloopstack=fl; // at least one
365          return false; // I think we need to exit now
366        } else {
367          if (verbose) { System.out.printf("NEXT: Looping back\n"); }
368          // goto the end of that for loop
369          executionpoint=forloopstack[fl];
370 
371          topforloopstack=fl+1; // new!!!
372
373          return true;
374        }
375      }
376    }
377    if (verbose) { System.out.printf("No more for loops to match!\n"); }
378
379    topforloopstack=fl; // new!!!
380
381    throw new BasicLineNotFoundError("NEXT WITHOUT FOR ERROR");
382  }

1918  int findLine(String lineno, boolean same, int pos[])
1919  {
1920    int at=0;
1921    boolean hasdigits=false;
1922    int start=0;
1923    int numstart=0;
1924  
1925    while (at<programText.length()) {
1926      hasdigits=false;
1927      while (at<programText.length()) {
1928        String a=programText.substring(at,at+1);
1929        if (!hasdigits) {
1930          if (a.equals("\n") || a.equals(" ")) { // chew these up
1931          } else if (a.compareTo("0")>=0 && a.compareTo("9")<=0) {
1932            numstart=at; // recording the actual start of the number
1933            hasdigits=true;
1934          } else {
1935            // didnt have a line number!
1936            hasdigits=false;
1937            break;
1938          }
1939        } else {
1940          if (!(a.compareTo("0")>=0 && a.compareTo("9")<=0)) {
1941            break;
1942          }
1943        }
1944        at++;
1945      } // end while finding line
1946      // read at spaces (do we need to)
1947      if (hasdigits) {
1948        // we have got number
1949        if (verbose) { System.out.printf(">> Line number %d to %d : %s\n",
1950          start,at,
1951          programText.substring(start,at)); }
1952        // we want to check the line #
1953        if (same && programText.substring(numstart,at).equals(lineno)) {
1954          if (verbose) System.out.printf("Exact match\n");
1955          pos[0]=start;
1956          return 0;
1957
1958        // must be a numeric compare!! (or equivalent)
1959        } else if (programText.substring(numstart,at).length()>lineno.length() ||
1960                     programText.substring(numstart,at).length()==lineno.length() 
1961                   && programText.substring(numstart,at).compareTo(lineno)>0) {
1962
1963          if (verbose) System.out.printf("Next higher match\n");
1964          pos[0]=start;
1965          return 1;
1966        }
1967      } 
1968   
1969      // read up until new line
1970      while (at<programText.length()) {
1971        String a=programText.substring(at,at+1);
1972        at++;
1973        if (a.equals("\n")) { start=at; break; }
1974      }
1975    }
1976    return -1;
1977  }
