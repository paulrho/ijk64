154   public void command_CIRCLE(int x1, int y1, int r, int col, int fill) {
155     newoffGraphics.setColor(colorindex[col]);
156     if (fill!=0) {
157       newoffGraphics.fillOval(x1,y1+tby,r,r);
158     } else {
159       newoffGraphics.drawOval(x1,y1+tby,r,r);
160     }
161     if (!inframe) doupdate();
162   }

164   public void command_RECT(int x1, int y1, int x2, int y2, int col) {
165     newoffGraphics.setColor(colorindex[col]);
166     newoffGraphics.fillRect(x1,y1+tby,x2-x1,y2-y1);
167     if (!inframe) doupdate();
168   }

170   public void command_FILL(int x1, int y1, int x2, int y2, int x3, int y3, int x4, int y4, int col) {
171     int xpoints[]= {x1, x2, x3, x4, x1};
172     int ypoints[]= {y1+tby, y2+tby, y3+tby, y4+tby, y1+tby};
173     if (col<16) 
174       newoffGraphics.setColor(colorindex[col]);
175     else
176       newoffGraphics.setColor(new Color(col));
177     newoffGraphics.fillPolygon(xpoints,ypoints,5);
178     if (!inframe) doupdate();
179   }

181   public void command_LINE(int x1, int y1, int x2, int y2, int col) {
182     newoffGraphics.setColor(colorindex[col]);
183     newoffGraphics.drawLine(x1,y1+tby,x2,y2+tby);
184     if (!inframe) doupdate();

192   public void command_LSET(int w, int c) {
193     if (c==4) {
194       if (w==0)   // 0,4 off
195         newoffGraphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
196       else        // 1,4 on
197         newoffGraphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
198     } else {
199       lsize=w;
200       newoffGraphics.setStroke(new BasicStroke(w));
201     }
202   }

219   public int command_PEEK(int address) {
220           if (address==1024) { return valevent; }
221           if (address==1030) { return valx&0xFF; }
222           if (address==1031) { return (valx>>8)&0xFF; }
223           if (address==1032) { return valy&0xFF; }
224           if (address==1033) { return (valy>>8)&0xFF; }
225           return 0;
226   }

250   public void command_DRAWIMAGE(int imgno, int x, int y, double scale, double rotation) {
251       AffineTransform tx = AffineTransform.getScaleInstance(scale, scale);
252//       tx.translate(x,y+(double)tby/scale); // was but scales the x,y too
253       tx.translate(x/scale,(y+(double)tby)/scale);
254       if (rotation!=0.0) tx.rotate(rotation); 
255       //AffineTransformOp op = new AffineTransformOp(tx, AffineTransformOp.TYPE_NEAREST_NEIGHBOR);
256       //newoffGraphics.drawImage(imgarray[imgno],tx,null);
257       AffineTransformOp op = new AffineTransformOp(tx, AffineTransformOp.TYPE_BICUBIC);
258       newoffGraphics.drawImage(imgarray[imgno],op,0,0);
259
260     if (!inframe) doupdate();
261   }

291   void dothings() {
292     // just in a loop
293
294     command_CLS();
295     while (true) {
296
297       // do out of paint drawing
298       
299       //newoffGraphics.drawLine(124,24,235,252);
300       command_LINE(
301         (int)(generator.nextDouble()*1000)
302         ,(int)(generator.nextDouble()*1000)
303         ,(int)(generator.nextDouble()*1000)
304         ,(int)(generator.nextDouble()*1000)
305         ,(int)(generator.nextDouble()*16)
306       );
307       command_GPRINT("x",(int)(generator.nextDouble()*1000)
308         ,(int)(generator.nextDouble()*1000),3);
309
310       if (command_PEEK(1024)!=0) {
311         command_GPRINT("m="+command_PEEK(1024),
312           command_PEEK(1030)+command_PEEK(1031)*256,
313           command_PEEK(1032)+command_PEEK(1033)*256
314           ,9
315         );
316         command_POKE(1024,0);
317           }
318         
319       doupdate();
320       try {
321         Thread.sleep(1);       // in the extreme slowdown
322       }
323       catch(InterruptedException e) {
324       }
325     }
326
327   }
