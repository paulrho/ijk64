308boolean ReadPart() throws BasicException
309{
310  // all new improved, it will return a couple of "global" codes
311  // indicating:
312  //   parttype
313  // we read from _pnt_ on the string _line_
314
315  // when changing String a= to chars, it took the same amount of time (in timingjar)
316
317      if (basictimer) { // here!!!!! save this
318        basictimer_thispnt=pnt;
319      }
320      if (speeder && machine.petspeed.pcache[pnt]>0) {
321        int nnn=machine.petspeed.pcache[pnt];
322        if (basictimer) 
323          basictimer_thispnt=machine.petspeed.btpnt[pnt];
324        pnt=machine.petspeed.pnext[pnt];
325        if (verbose) System.out.printf("found a cached line start nnn=%d(1=assign) pnt=%d\n",nnn,pnt);
326        if (nnn==1) {
327          partType=PT_ASSIGN;
328          return true;
329        } else {
330          partType=PT_TOKEN;
331          gotToken=nnn-2;
332          return true;
333        }
334      }
335    int fpnt=pnt; // speeder - first appearence
336    while (pnt<linelength) {
337      String a=line.substring(pnt,pnt+1);
338      //if (partType==PT_NEWLINE && a.equals("\r")) continue; // try this to allow CRLF
339      //else 
340      if (partType==PT_NEWLINE && a.compareTo("0")>=0 && a.compareTo("9")<=0) {
341        // just read it here like the token thing
342if (dofulltiming) { start_timing(TIME_ReadLineNo); }
343        if (ReadLineNo()) {
344if (dofulltiming) { end_timing(TIME_ReadLineNo); }
345          partType=PT_HASLINE; // we are up to statements
346          // pnt+=at+1; already done (+1 is good?)
347          // keepLine set to line # (string)
348          // but wait, lets keep going...
349          if (verbose) { System.out.printf("On Line %s\n",keepLine); }
350          // not for now /// machine.currentLineNo=keepLine; // only added for error tracking - might slow things down
351          if (basictimer) { // here!!!!! save this
352            basictimer_thispnt=pnt;
353          }
354
355        } else {
356          // we have a problem
357          System.out.printf("?SYNTAX ERROR parsing lineno\n");
358          throw new BasicException("SYNTAX ERROR : PARSING LINENO : NEEDS A SPACE");
359          // return false;
360        }
361      } else if (a.compareTo("?")==0) {
362        // special case print token
363        pnt++;
364        partType=PT_TOKEN;
365        gotToken=ST_PRINT;
366        if (speeder) {
367          machine.petspeed.pcache[fpnt]=gotToken+2;
368          if (basictimer) 
369            machine.petspeed.btpnt[fpnt]=basictimer_thispnt;
370          machine.petspeed.pnext[fpnt]=pnt;
371          if (verbose) System.out.printf("caching a print\n");
372        }
373        return true;
374      } else if (a.compareToIgnoreCase("a")>=0 && a.compareTo("z")<=0) {
375        // one of two things here - a token, or an assignment!
376if (dofulltiming) { start_timing(TIME_ReadStatementToken); }
377        if (ReadStatementToken()) {
378if (dofulltiming) { end_timing(TIME_ReadStatementToken); }
379          partType=PT_TOKEN;
380          // gotToken already set
381          // pnt+=at; already done
382          if (gotToken==ST_DEF || gotToken==ST_LET) { // this adds a bit EVERY statement...
383            if (ReadStatementToken()) {
384              partType=PT_TOKEN;
385              return true;
386            } else {
387              partType=PT_ASSIGN;
388              return true;
389            }
390          }
391          if (speeder) {
392            machine.petspeed.pcache[fpnt]=gotToken+2;
393            if (basictimer) 
394              machine.petspeed.btpnt[fpnt]=basictimer_thispnt;
395            machine.petspeed.pnext[fpnt]=pnt;
396            if (verbose) System.out.printf("caching a token\n");
397          }
398          return true;
399        } else {
400if (dofulltiming) { end_timing(TIME_ReadStatementToken); }
401          // have to assume this is an assignment
402          partType=PT_ASSIGN;
403          // we dont move pnt at all, we start from the start!
404          if (speeder) {
405            machine.petspeed.pcache[fpnt]=1;
406            if (basictimer) 
407              machine.petspeed.btpnt[fpnt]=basictimer_thispnt;
408            machine.petspeed.pnext[fpnt]=pnt;
409            if (verbose) System.out.printf("caching an assign\n");
410          }
411          return true;
412        }
413      } else if (a.equals("\n")) {
414        partType=PT_NEWLINE;
415        // keep going!
416        pnt++;
417      } else if (a.equals(":")) {
418        pnt++; 
419      } else if (a.equals(" ")) { // or TAB
420        pnt++; // no, chew it up
421      } else {
422        // we have found something that finished the sequence
423        // we have NO idea what is this!
424        System.out.printf("?SYNTAX ERROR parsing got character \"%s\"\n",a);
425        return false;
426      }
427    }
428    // is this really an error? (it is the end!)
429    // System.out.printf("?SYNTAX ERROR parsing - we fell off the end\n");
430    // return false;
431    partType=PT_EOF;
432    return true;
433} // ReadPart

490                        // read out rest of line

492                        String a;
493                        while (pnt<linelength && !(a=line.substring(pnt,pnt+1)).equals("\n")) {
494                          building+=a;
495                          pnt++;
496                        }
497                        machine.cacheDataAdd(building+"\n");
498                        if (verbose) { System.out.printf("Adding %s\n",building); }
499                  }
500                  if (gotToken==ST_LABEL) {
501                        SkipSpaces();
502                        if (verbose) { System.out.printf("Found a label\n"); }
503                        // from here to the end of the line is to be cached
504                        // read out rest of line
505                        String building="";
506                        String a;
507                        while (pnt<linelength && !(a=line.substring(pnt,pnt+1)).equals("\n") && !(a=line.substring(pnt,pnt+1)).equals(":")) {
508                          building+=a;
509                          pnt++;
510                        }
511                        // trim off ending spaces (but allow middle spaces)
512                        machine.cacheLabel(building.trim(),pnt); //cache the label - not the line
513                        if (verbose) { System.out.printf("Label %s\n",building); }
514                  }
515                }
516    }
517    IgnoreRestofLine();
518
519    if (pnt>=line.length()) {
520      break;
521    }
522    SkipNewLines();
523  }
