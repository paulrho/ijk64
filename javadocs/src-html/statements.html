307boolean ReadPart() throws BasicException
308{
309  // all new improved, it will return a couple of "global" codes
310  // indicating:
311  //   parttype
312  // we read from _pnt_ on the string _line_
313
314  // when changing String a= to chars, it took the same amount of time (in timingjar)
315
316          if (basictimer) { // here!!!!! save this
317            basictimer_thispnt=pnt;
318          }
319    while (pnt<linelength) {
320      String a=line.substring(pnt,pnt+1);
321      //if (partType==PT_NEWLINE && a.equals("\r")) continue; // try this to allow CRLF
322      //else 
323      if (partType==PT_NEWLINE && a.compareTo("0")>=0 && a.compareTo("9")<=0) {
324        // just read it here like the token thing
325if (dofulltiming) { start_timing(TIME_ReadLineNo); }
326        if (ReadLineNo()) {
327if (dofulltiming) { end_timing(TIME_ReadLineNo); }
328          partType=PT_HASLINE; // we are up to statements
329          // pnt+=at+1; already done (+1 is good?)
330          // keepLine set to line # (string)
331          // but wait, lets keep going...
332          if (verbose) { System.out.printf("On Line %s\n",keepLine); }
333          // not for now /// machine.currentLineNo=keepLine; // only added for error tracking - might slow things down
334          if (basictimer) { // here!!!!! save this
335            basictimer_thispnt=pnt;
336          }
337
338        } else {
339          // we have a problem
340          System.out.printf("?SYNTAX ERROR parsing lineno\n");
341          throw new BasicException("SYNTAX ERROR : PARSING LINENO : NEEDS A SPACE");
342          // return false;
343        }
344      } else if (a.compareTo("?")==0) {
345        // special case print token
346        pnt++;
347        partType=PT_TOKEN;
348        gotToken=ST_PRINT;
349        return true;
350      } else if (a.compareToIgnoreCase("a")>=0 && a.compareTo("z")<=0) {
351        // one of two things here - a token, or an assignment!
352if (dofulltiming) { start_timing(TIME_ReadStatementToken); }
353        if (ReadStatementToken()) {
354if (dofulltiming) { end_timing(TIME_ReadStatementToken); }
355          partType=PT_TOKEN;
356          // gotToken already set
357          // pnt+=at; already done
358          if (gotToken==ST_DEF || gotToken==ST_LET) { // this adds a bit EVERY statement...
359            if (ReadStatementToken()) {
360              partType=PT_TOKEN;
361              return true;
362            } else {
363              partType=PT_ASSIGN;
364              return true;
365            }
366          }
367          return true;
368        } else {
369if (dofulltiming) { end_timing(TIME_ReadStatementToken); }
370          // have to assume this is an assignment
371          partType=PT_ASSIGN;
372          // we dont move pnt at all, we start from the start!
373          return true;
374        }
375      } else if (a.equals("\n")) {
376        partType=PT_NEWLINE;
377        // keep going!
378        pnt++;
379      } else if (a.equals(":")) {
380        pnt++; 
381      } else if (a.equals(" ")) { // or TAB
382        pnt++; // no, chew it up
383      } else {
384        // we have found something that finished the sequence
385        // we have NO idea what is this!
386        System.out.printf("?SYNTAX ERROR parsing got character \"%s\"\n",a);
387        return false;
388      }
389    }
390    // is this really an error? (it is the end!)
391    // System.out.printf("?SYNTAX ERROR parsing - we fell off the end\n");
392    // return false;
393    partType=PT_EOF;
394    return true;
395} // ReadPart
