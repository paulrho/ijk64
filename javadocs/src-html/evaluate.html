889    oper=oper.toLowerCase(); //20060204pgs

903         if (oper.equals("or")) { answer=(double)(((int)(left)) | ((int)(right))); }

930    if (verbose) { System.out.printf("%sCalculating %f %s %f, Answer = %f\n",printprefix,left,oper,right,answer); }
931    if (verbose) { show_state(); }
932    return answer;
933  }

987    if (verbose) { System.out.printf("%sGot %s oper\n",printprefix,op); }
