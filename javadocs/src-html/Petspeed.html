007{
008  static final int MAX=5000;
009  // store program memory
010  int top=1; // because 0 is a special pointer to show invalid
011  int tmptop;
012  boolean record=false;
013  int []prog= new int[MAX]; // program memory instruction
014  double [] pargD = new double[MAX]; // double arg
015  int [] pargmem = new int[MAX]; // mem pointer arg
016  String [] pargS = new String[MAX]; // string arg
017  Machine using_machine=null;
018
019  static final int MAXPSIZE=50000; // just for now, have a more efficient sparse array later
020  int []acpointer = new int[MAXPSIZE];
021  int []acpointer_next = new int[MAXPSIZE];

023  Petspeed(Machine m) {
024    using_machine=m;
025  }
026  // storing the program
027  int getTop() { return top; }
028  int addInstr(int instr) { return addInstr(instr, 0.0,-1, ""); }
029  int addInstr(int instr, String argS) { return addInstr(instr, 0.0,-1, argS); }
030  int addInstr(int instr, double argD, int argmem) { return addInstr(instr, argD, argmem, ""); }
031  int addInstr(int instr, double argD) { return addInstr(instr, argD, -1, ""); }
032  int addInstr(int instr, int argmem)  { return addInstr(instr, 0.0, argmem, ""); }
033  int addInstr(int instr, double argD, int argmem, String argS) {
034    if (record) {
035      prog[tmptop]=instr;
036      pargD[tmptop]=argD;
037      pargmem[tmptop]=argmem;
038      pargS[tmptop]=argS;
039      if (using_machine.verbose) System.out.printf("%d %d %d %f %s\n",tmptop,prog[tmptop],pargmem[tmptop],pargD[tmptop],pargS[tmptop]);
040      //if (instr<0) System.out.printf("-1 INSTRUCTION\n"); // debug FIX
041      tmptop++;
042      return 1;
043    } else return 0;
044  }
045  void rewind() {
046    tmptop--;
047  }

049  static int ftoken(String f) {
050    for (int i=0; i<O_strings.length; ++i)
051      if (O_strings[i].equals(f)) return i;
052    for (int i=0; i<F_strings.length; ++i)
053      if (F_strings[i].equals(f)) return i;
054    //if (using_machine.verbose) System.out.printf("A-COMPILER could not find %s (okay if array)\n",f);
055    return -1;
056  }

061  static final int AMAX=100;
062  double astack_d[]=new double[AMAX];
063  String astack_s[]=new String[AMAX];
064  int atop=0;
065  java.text.SimpleDateFormat localDateFormat = new java.text.SimpleDateFormat("HHmmss"); // for TI$ efficiency
066
067  int execute(int x) throws EvaluateException {
068    boolean verbose=using_machine.verbose;
069    for (int i=acpointer[x]; i<MAX; ++i) {
070      //if (prog[i]==I_HLT) break;
071      if (verbose) System.out.printf("EXECUTING %d %d %d %f %s  ",i,prog[i],pargmem[i],pargD[i],pargS[i]);
072      switch(prog[i]) {
073        case I_HLT: 
074          if (verbose) System.out.printf("\n");
075          if (atop!=0 && atop!=1) System.out.printf("atop not at zero or one\n");
076          return nextpnt(x);
077        case I_FNC | F_sin : 
078          astack_d[atop-1]= Math.sin(astack_d[atop-1]);
079          break;
080        case I_FNC | F_cos : 
081          astack_d[atop-1]= Math.cos(astack_d[atop-1]);
082          break;
083        case I_FNC | F_int : 
084          astack_d[atop-1]= (double)((int)(astack_d[atop-1]));
085          break;
086        case I_FNC | F_log : 
087          astack_d[atop-1]= Math.log(astack_d[atop-1]);
088          break;
089        case I_FNC | F_tan : 
090          astack_d[atop-1]= Math.tan(astack_d[atop-1]);
091          break;
092        case I_FNC | F_atn : 
093          astack_d[atop-1]= Math.atan(astack_d[atop-1]);
094          break;
095        case I_FNC | F_sqr : 
096        case I_FNC | F_sqrt : 
097          astack_d[atop-1]= Math.sqrt(astack_d[atop-1]);
098          break;
099        case I_FNC | F_asin : 
100          astack_d[atop-1]= Math.asin(astack_d[atop-1]);
101          break;
102        case I_FNC | F_acos : 
103          astack_d[atop-1]= Math.acos(astack_d[atop-1]);
104          break;
105        case I_FNC | F_abs : 
106          astack_d[atop-1]= Math.abs(astack_d[atop-1]);
107          break;
108        case I_FNC | F_rnd : 
109          if (astack_d[atop-1]<0) { using_machine.evaluate_engine.generator = new java.util.Random((long)-astack_d[atop-1]); } // added this in for seeds
110          astack_d[atop-1]=using_machine.evaluate_engine.generator.nextDouble();
111          break;
112        case I_FNC | F_exp : 
113          astack_d[atop-1]= Math.exp(astack_d[atop-1]);
114          break;
115        case I_FNC | F_sgn : 
116          if (astack_d[atop-1]<0.0) astack_d[atop-1]=-1.0;
117          else if (astack_d[atop-1]==0.0) astack_d[atop-1]=0.0;
118          else astack_d[atop-1]=1.0;
119          break;
120
121        case I_FNC | F_val : 
122          // takes a string but returns a double
123          try {
124            astack_d[atop-1]=Double.parseDouble(astack_s[atop-1]); 
125          } catch (Exception e) { 
126            // maybe just a "." should be coverted to 0 or a ""???
127            // throw new EvaluateException("NON NUMERIC STRING"); // look at what the C64 really does
128            astack_d[atop-1]=0.0;
129          }
130          break;
131
132        case I_FNC | F_midD : 
133          try {          
134            astack_s[atop-3]=astack_s[atop-3].substring((int)astack_d[atop-2]-1,(int)astack_d[atop-2]-1+(int)astack_d[atop-1]);
135            atop--;
136            atop--;
137          } catch(Exception e) {
138            throw new EvaluateException("BAD SUBSTRING INDEX");              
139          }
140          break;
141        case I_FNC | F_leftD : 
142          try {          
143            astack_s[atop-2]=astack_s[atop-2].substring(0,(int)astack_d[atop-1]);
144            atop--;
145          } catch(Exception e) {
146            throw new EvaluateException("BAD SUBSTRING INDEX");              
147          }
148          break;
149        case I_FNC | F_rightD : 
150          try {          
151            astack_s[atop-2]=astack_s[atop-2].substring(astack_s[atop-2].length()-(int)astack_d[atop-1]);
152            atop--;
153          } catch(Exception e) {
154            throw new EvaluateException("BAD SUBSTRING INDEX");              
155          }
156          break;
157        case I_FNC | F_strD : 
158          // I think the leading space is wrong - it should be -ve if it is.... FIX
159          if (astack_d[atop-1]-(int)astack_d[atop-1]==0.0) {
160            astack_s[atop-1]=" "+(int)astack_d[atop-1];
161          } else {
162            astack_s[atop-1]=" "+(new Double(astack_d[atop-1]).toString());
163          }
164          break;
165        case I_FNC | F_len : 
166          try {          
167            astack_d[atop-1]=astack_s[atop-1].length();
168          } catch(Exception e) {
169            throw new EvaluateException("BAD SUBSTRING INDEX");              
170          }
171          break;
172
173        case I_FNC | F_peek : 
174          astack_d[atop-1]=using_machine.peek((int)astack_d[atop-1]);
175          break;
176        case I_FNC | F_pos : 
177          {
178            int cursx=using_machine.machinescreen.cursX;
179            astack_d[atop-1]=cursx;
180          }
181          break;
182
183        case I_FNC | F_asc : 
184          {
185            if (astack_s[atop-1]=="") throw new EvaluateException("ILLEGAL QUANTITY");
186            int v=(int)((astack_s[atop-1].charAt(0))&0xFF);
187            if (v>=65&&v<=90) v+=128; 
188            else if (v>=65&&v<=90) v+=32;
189            else if (v>=97&&v<=122) v-=32; 
190            else if (v>=193&&v<=218) v-=96; 
191            else if (v==95) v=96; 
192            else if (v==96) v=95; 
193            else if (v==123) v=179; else if (v==179) v=123;
194            else if (v==124) v=125; 
195            else if (v==125) v=171; 
196            else if (v==171) v=124;  
197            astack_d[atop-1]=v;
198          }
199          break;
200        case I_FNC | F_chrD : 
201          {
202            int v=(int)astack_d[atop-1];
203            if (v>=193&&v<=218) v=v-128; 
204            else if (v>=65&&v<=90) v=v+32; 
205            else if (v>=97&&v<=122) v=v+96; 
206            else if (v==95) v=96; 
207            else if (v==96) v=95; 
208            else if (v==123) v=179; else if (v==179) v=123;
209            else if (v==124) v=171; 
210            else if (v==125) v=124; 
211            else if (v==171) v=125;  
212            astack_s[atop-1]=""+(char)v;
213          }
214          break;
215
216        case I_FNC | F_var_ti : 
217          astack_d[atop++]= (double)(int)((System.currentTimeMillis()/16.66666666)%1073741824);
218          break;
219        case I_FNC | F_var_st : 
220          astack_d[atop++]= 0; // FIX
221          break;
222        case I_FNC | F_var_tiD : 
223          astack_s[atop++]= localDateFormat.format( System.currentTimeMillis());
224          break;
225        case I_FNC | F_var_pi : 
226          astack_d[atop++]= Math.PI;
227          break;
228
229        case I_PRF | T_Str | O_add : 
230          astack_s[atop-2]= astack_s[atop-2] + astack_s[atop-1]; atop--;
231          break;
232
233        case I_PRF | O_pow : 
234          astack_d[atop-2]= Math.pow(astack_d[atop-2],astack_d[atop-1]); atop--;
235          break;
236        case I_PRF | O_neg : 
237          astack_d[atop-1]= -astack_d[atop-1];
238          break;
239        case I_PRF | O_mul : 
240          astack_d[atop-2]= astack_d[atop-2] * astack_d[atop-1]; atop--;
241          break;
242        case I_PRF | O_add : 
243          astack_d[atop-2]= astack_d[atop-2] + astack_d[atop-1]; atop--;
244          break;
245        case I_PRF | O_sub : 
246          astack_d[atop-2]= astack_d[atop-2] - astack_d[atop-1]; atop--;
247          break;
248        case I_PRF | O_div : 
249          astack_d[atop-2]= astack_d[atop-2] / astack_d[atop-1]; atop--;
250          break;
251        case I_PRF | O_lt : 
252          astack_d[atop-2]= (astack_d[atop-2] < astack_d[atop-1])?-1:0; atop--;
253          break;
254        case I_PRF | O_le : 
255          astack_d[atop-2]= (astack_d[atop-2] <= astack_d[atop-1])?-1:0; atop--;
256          break;
257        case I_PRF | O_eq : 
258          astack_d[atop-2]= (astack_d[atop-2] == astack_d[atop-1])?-1:0; atop--;
259          break;
260        case I_PRF | O_ge : 
261        case I_PRF | O_ge2 : 
262          astack_d[atop-2]= (astack_d[atop-2] >= astack_d[atop-1])?-1:0; atop--;
263          break;
264        case I_PRF | O_gt : 
265          astack_d[atop-2]= (astack_d[atop-2] > astack_d[atop-1])?-1:0; atop--;
266          break;
267        case I_PRF | O_ne : 
268          astack_d[atop-2]= (astack_d[atop-2] != astack_d[atop-1])?-1:0; atop--;
269          break;
270         //if (oper.equals("or")) { answer=(double)(((int)(left)) | ((int)(right))); }
271    //else if (oper.equals("xor")) { answer=(double)(((int)(left)) ^ ((int)(right))); }
272    //else if (oper.equals("and")) { answer=(double)(((int)(left)) & ((int)(right))); }
273    //else if (oper.equals("not")) { answer=(double)(~(int)(right)); }
274        case I_PRF | O_not : 
275          astack_d[atop-1]= (double)(~(int)(astack_d[atop-1]));
276          break;
277        case I_PRF | O_and : 
278          astack_d[atop-2]= (((int)(astack_d[atop-2])) & ((int)(astack_d[atop-1]))); atop--;
279          break;
280        case I_PRF | O_or : 
281          astack_d[atop-2]= (((int)(astack_d[atop-2])) | ((int)(astack_d[atop-1]))); atop--;
282          break;
283        case I_PRF | O_xor : 
284          astack_d[atop-2]= (((int)(astack_d[atop-2])) ^ ((int)(astack_d[atop-1]))); atop--;
285          break;
286
287        case I_PRF | T_Str | O_gt : 
288          astack_d[atop-2]= (astack_s[atop-2].compareTo(astack_s[atop-1])>0)?-1:0; atop--;
289          break;
290        case I_PRF | T_Str | O_lt : 
291          astack_d[atop-2]= (astack_s[atop-1].compareTo(astack_s[atop-2])>0)?-1:0; atop--;
292          break;
293        case I_PRF | T_Str | O_ne : 
294          astack_d[atop-2]= (astack_s[atop-2].equals(astack_s[atop-1]))?0:-1; atop--;
295          break;
296        case I_PRF | T_Str | O_eq : 
297          astack_d[atop-2]= (astack_s[atop-2].equals(astack_s[atop-1]))?-1:0; atop--;
298          break;
299        case I_PSH | T_Str | M_IMM : 
300          astack_s[atop++]=pargS[i];
301          break;
302        case I_PSH | T_Str | M_MEM : 
303          astack_s[atop++]=using_machine.variables.variablestring[pargmem[i]];
304          break;
305        case I_PSH | T_Str | M_MEMARR1 : 
306          astack_s[atop-1]=using_machine.variables.variablearraystring1[pargmem[i]][(int)astack_d[atop-1]];
307          break;
308        case I_PSH | T_Str | M_MEMARR2 : 
309          astack_s[atop-2]=using_machine.variables.variablearraystring2[pargmem[i]][(int)astack_d[atop-2]][(int)astack_d[atop-1]]; atop--;
310          break;
311        case I_STO | T_Str | M_MEM : 
312          using_machine.variables.variablestring[pargmem[i]]=astack_s[--atop];
313          break;
314        case I_STO | T_Str | M_MEMARR1 : 
315          using_machine.variables.variablearraystring1[pargmem[i]][(int)astack_d[atop-2]]=astack_s[atop-1];
316          atop--; atop--;
317          break;
318        case I_STO | T_Str | M_MEMARR2 : 
319          using_machine.variables.variablearraystring2[pargmem[i]][(int)astack_d[atop-3]][(int)astack_d[atop-2]]=astack_s[atop-1];
320          atop--; atop--; atop--;
321          break;
322
323        case I_PSH | T_Dbl | M_IMM : 
324          astack_d[atop++]=pargD[i];
325          break;
326        case I_PSH | T_Dbl | M_MEM : 
327          astack_d[atop++]=using_machine.variables.variablevalue[pargmem[i]];
328          break;
329        case I_PSH | T_Dbl | M_MEMARR1 : 
330          astack_d[atop-1]=using_machine.variables.variablearrayvalue1[pargmem[i]][(int)astack_d[atop-1]];
331          break;
332        case I_PSH | T_Dbl | M_MEMARR2 : 
333          astack_d[atop-2]=using_machine.variables.variablearrayvalue2[pargmem[i]][(int)astack_d[atop-2]][(int)astack_d[atop-1]];
334          atop--;
335          break;
336        case I_STO | T_Dbl | M_MEM : 
337          using_machine.variables.variablevalue[pargmem[i]]=astack_d[--atop];
338          break;
339        case I_STO | T_Dbl | M_MEMARR1 : 
340          using_machine.variables.variablearrayvalue1[pargmem[i]][(int)astack_d[atop-2]]=astack_d[atop-1];
341          atop--; atop--;
342          break;
343        case I_STO | T_Dbl | M_MEMARR2 : 
344          using_machine.variables.variablearrayvalue2[pargmem[i]][(int)astack_d[atop-3]][(int)astack_d[atop-2]]=astack_d[atop-1];
345          atop--; atop--; atop--;
346          break;
347        default:
348          if (verbose) System.out.printf("X ");
349          System.out.printf("Instruction Fault at %d instruction %d\n",i,prog[i]);
350          // should through an error! Instruction Fault
351          throw new EvaluateException("INSTRUCTION FAULT");              
352          //break;
353      }
354      if (verbose) System.out.printf("\n");
355    }
356    //return nextpnt(x); // we've gone off the end
357    return -1;
358  }

390  int savestart_p=0;

392  boolean is_compiled(int x) { return acpointer[x]!=0; }
393  void savestart(int x) { savestart_p=x; savestart_ac=top; tmptop=top; record=true; }
394  void saveacode(int end) 
395  {
396          // push a HLT onto code
397          addInstr(I_HLT); 
398          acpointer[savestart_p]=savestart_ac;
399          acpointer_next[savestart_p]=end;
400          // this is now marked as a valid piece of code
401          top=tmptop;
402          if (using_machine.verbose) System.out.printf("Saved code from %d to %d for location %d\n",savestart_ac,top,savestart_p);
403          record=false;
404  }
