036  int addInstr(int instr, double argD, int argmem, String argS) {
037    if (record) {
038      prog[tmptop]=instr;
039      pargD[tmptop]=argD;
040      pargmem[tmptop]=argmem;
041      pargS[tmptop]=argS;
042      if (using_machine.verbose) System.out.printf("%d %d %d %f %s\n",tmptop,prog[tmptop],pargmem[tmptop],pargD[tmptop],pargS[tmptop]);
043      //if (instr<0) System.out.printf("-1 INSTRUCTION\n"); // debug FIX
044      tmptop++;
045      return 1;
046    } else return 0;
047  }
048  void rewind() {
049    tmptop--;
050  }

052  static int ftoken(String f) {
053    for (int i=0; i<O_strings.length; ++i)
054      if (O_strings[i].equals(f)) return i;
055    for (int i=0; i<F_strings.length; ++i)
056      if (F_strings[i].equals(f)) return i;
057    //if (using_machine.verbose) System.out.printf("A-COMPILER could not find %s (okay if array)\n",f);
058    return -1;
059  }

070  int execute(int x) throws EvaluateException {
071    try { // safety catch!
072    boolean verbose=using_machine.verbose;
073    for (int i=acpointer[x]; i<MAX; ++i) {
074      //if (prog[i]==I_HLT) break;
075      if (verbose) System.out.printf("EXECUTING %d %d %d %f %s  ",i,prog[i],pargmem[i],pargD[i],pargS[i]);
076      switch(prog[i]) {
077        case I_HLT: 
078          if (verbose) System.out.printf("\n");
079          if (atop!=0 && atop!=1) System.out.printf("atop not at zero or one\n");
080          return nextpnt(x);
081        case I_FNC | F_sin : 
082          astack_d[atop-1]= Math.sin(astack_d[atop-1]);
083          break;
084        case I_FNC | F_cos : 
085          astack_d[atop-1]= Math.cos(astack_d[atop-1]);
086          break;
087        case I_FNC | F_int : 
088          astack_d[atop-1]= (double)((int)(astack_d[atop-1]));
089          break;
090        case I_FNC | F_log : 
091          astack_d[atop-1]= Math.log(astack_d[atop-1]);
092          break;
093        case I_FNC | F_tan : 
094          astack_d[atop-1]= Math.tan(astack_d[atop-1]);
095          break;
096        case I_FNC | F_atn : 
097          astack_d[atop-1]= Math.atan(astack_d[atop-1]);
098          break;
099        case I_FNC | F_sqr : 
100        case I_FNC | F_sqrt : 
101          astack_d[atop-1]= Math.sqrt(astack_d[atop-1]);
102          break;
103        case I_FNC | F_asin : 
104          astack_d[atop-1]= Math.asin(astack_d[atop-1]);
105          break;
106        case I_FNC | F_acos : 
107          astack_d[atop-1]= Math.acos(astack_d[atop-1]);
108          break;
109        case I_FNC | F_abs : 
110          astack_d[atop-1]= Math.abs(astack_d[atop-1]);
111          break;
112        case I_FNC | F_rnd : 
113          if (astack_d[atop-1]<0) { using_machine.evaluate_engine.generator = new java.util.Random((long)-astack_d[atop-1]); } // added this in for seeds
114          astack_d[atop-1]=using_machine.evaluate_engine.generator.nextDouble();
115          break;
116        case I_FNC | F_exp : 
117          astack_d[atop-1]= Math.exp(astack_d[atop-1]);
118          break;
119        case I_FNC | F_sgn : 
120          if (astack_d[atop-1]<0.0) astack_d[atop-1]=-1.0;
121          else if (astack_d[atop-1]==0.0) astack_d[atop-1]=0.0;
122          else astack_d[atop-1]=1.0;
123          break;
124
125        case I_FNC | F_val : 
126          // takes a string but returns a double
127          try {
128            astack_d[atop-1]=Double.parseDouble(astack_s[atop-1]); 
129          } catch (Exception e) { 
130            // maybe just a "." should be coverted to 0 or a ""???
131            // throw new EvaluateException("NON NUMERIC STRING"); // look at what the C64 really does
132            astack_d[atop-1]=0.0;
133          }
134          break;
135
136        case I_FNC | F_instr : 
137          astack_d[atop-2]=0;
138          for (int j=0; j<=astack_s[atop-2].length()-astack_s[atop-1].length(); ++j) {
139            if (astack_s[atop-2].substring(j,j+astack_s[atop-1].length()).equals(astack_s[atop-1])) {
140              astack_d[atop-2]=j+1;
141              break;
142            }
143          }
144          atop--;
145          break;
146
147        case I_FNC | F_frm : 
148          try {          
149            astack_s[atop-2]=String.format(astack_s[atop-2],astack_d[atop-1]);
150            atop--;
151          } catch(Exception e) {
152            throw new EvaluateException("BAD FORMAT");              
153          }
154          break;
155        case I_FNC | F_midD : 
156          try {          
157            astack_s[atop-3]=astack_s[atop-3].substring((int)astack_d[atop-2]-1,(int)astack_d[atop-2]-1+(int)astack_d[atop-1]);
158            atop--;
159            atop--;
160          } catch(Exception e) {
161            throw new EvaluateException("BAD SUBSTRING INDEX");              
162          }
163          break;
164        case I_FNC | F_leftD : 
165          //try {          
166            //astack_s[atop-2]=astack_s[atop-2].substring(0,(int)astack_d[atop-1]);
167
168            if ((int)astack_d[atop-1]<=0) { astack_s[atop-2]=""; }
169            else if ((int)astack_d[atop-1]>astack_s[atop-2].length()) { }
170            else { astack_s[atop-2]=astack_s[atop-2].substring(0,(int)astack_d[atop-1]); }
171            atop--;
172          //} catch(Exception e) {
173            //throw new EvaluateException("BAD SUBSTRING INDEX");              
174          //}
175          break;
176        case I_FNC | F_rightD : 
177          try {          
178            astack_s[atop-2]=astack_s[atop-2].substring(astack_s[atop-2].length()-(int)astack_d[atop-1]);
179            atop--;
180          } catch(Exception e) {
181            throw new EvaluateException("BAD SUBSTRING INDEX");              
182          }
183          break;
184        case I_FNC | F_strD : 
185          // I think the leading space is wrong - it should be -ve if it is.... FIX
186          if (astack_d[atop-1]-(int)astack_d[atop-1]==0.0) {
187            astack_s[atop-1]=" "+(int)astack_d[atop-1];
188          } else {
189            astack_s[atop-1]=" "+(new Double(astack_d[atop-1]).toString());
190          }
191          break;
192        case I_FNC | F_len : 
193          try {          
194            astack_d[atop-1]=astack_s[atop-1].length();
195          } catch(Exception e) {
196            throw new EvaluateException("BAD SUBSTRING INDEX");              
197          }
198          break;
199
200        case I_FNC | F_peek : 
201          astack_d[atop-1]=using_machine.peek((int)astack_d[atop-1]);
202          break;
203        case I_FNC | F_pos : 
204          {
205            int cursx=using_machine.machinescreen.cursX;
206            astack_d[atop-1]=cursx;
207          }
208          break;
209
210        case I_FNC | F_asc : 
211          {
212            if (astack_s[atop-1]=="") throw new EvaluateException("ILLEGAL QUANTITY");
213            int v=(int)((astack_s[atop-1].charAt(0))&0xFF);
214            if (v>=65&&v<=90) v+=128; 
215            else if (v>=65&&v<=90) v+=32;
216            else if (v>=97&&v<=122) v-=32; 
217            else if (v>=193&&v<=218) v-=96; 
218            else if (v==95) v=96; 
219            else if (v==96) v=95; 
220            else if (v==123) v=179; else if (v==179) v=123;
221            else if (v==124) v=125; 
222            else if (v==125) v=171; 
223            else if (v==171) v=124;  
224            astack_d[atop-1]=v;
225          }
226          break;
227        case I_FNC | F_chrD : 
228          {
229            int v=(int)astack_d[atop-1];
230            if (v>=193&&v<=218) v=v-128; 
231            else if (v>=65&&v<=90) v=v+32; 
232            else if (v>=97&&v<=122) v=v+96; 
233            else if (v==95) v=96; 
234            else if (v==96) v=95; 
235            else if (v==123) v=179; else if (v==179) v=123;
236            else if (v==124) v=171; 
237            else if (v==125) v=124; 
238            else if (v==171) v=125;  
239            astack_s[atop-1]=""+(char)v;
240          }
241          break;
242
243        case I_FNC | F_var_ti : 
244          astack_d[atop++]= (double)(int)((System.currentTimeMillis()/16.66666666)%1073741824);
245          break;
246        case I_FNC | F_var_st : 
247          astack_d[atop++]= 0; // FIX
248          break;
249        case I_FNC | F_var_tiD : 
250          astack_s[atop++]= localDateFormat.format( System.currentTimeMillis());
251          break;
252        case I_FNC | F_var_pi : 
253          astack_d[atop++]= Math.PI;
254          break;
255
256        case I_PRF | T_Str | O_add : 
257          astack_s[atop-2]= astack_s[atop-2] + astack_s[atop-1]; atop--;
258          break;
259
260        case I_PRF | O_pow : 
261          astack_d[atop-2]= Math.pow(astack_d[atop-2],astack_d[atop-1]); atop--;
262          break;
263        case I_PRF | O_neg : 
264          astack_d[atop-1]= -astack_d[atop-1];
265          break;
266        case I_PRF | O_mul : 
267          astack_d[atop-2]= astack_d[atop-2] * astack_d[atop-1]; atop--;
268          break;
269        case I_PRF | O_add : 
270          astack_d[atop-2]= astack_d[atop-2] + astack_d[atop-1]; atop--;
271          break;
272        case I_PRF | O_sub : 
273          astack_d[atop-2]= astack_d[atop-2] - astack_d[atop-1]; atop--;
274          break;
275        case I_PRF | O_div : 
276          astack_d[atop-2]= astack_d[atop-2] / astack_d[atop-1]; atop--;
277          break;
278        case I_PRF | O_lt : 
279          astack_d[atop-2]= (astack_d[atop-2] < astack_d[atop-1])?-1:0; atop--;
280          break;
281        case I_PRF | O_le : 
282          astack_d[atop-2]= (astack_d[atop-2] <= astack_d[atop-1])?-1:0; atop--;
283          break;
284        case I_PRF | O_eq : 
285          astack_d[atop-2]= (astack_d[atop-2] == astack_d[atop-1])?-1:0; atop--;
286          break;
287        case I_PRF | O_ge : 
288        case I_PRF | O_ge2 : 
289          astack_d[atop-2]= (astack_d[atop-2] >= astack_d[atop-1])?-1:0; atop--;
290          break;
291        case I_PRF | O_gt : 
292          astack_d[atop-2]= (astack_d[atop-2] > astack_d[atop-1])?-1:0; atop--;
293          break;
294        case I_PRF | O_ne : 
295          astack_d[atop-2]= (astack_d[atop-2] != astack_d[atop-1])?-1:0; atop--;
296          break;
297         //if (oper.equals("or")) { answer=(double)(((int)(left)) | ((int)(right))); }
298    //else if (oper.equals("xor")) { answer=(double)(((int)(left)) ^ ((int)(right))); }
299    //else if (oper.equals("and")) { answer=(double)(((int)(left)) & ((int)(right))); }
300    //else if (oper.equals("not")) { answer=(double)(~(int)(right)); }
301        case I_PRF | O_not : 
302          astack_d[atop-1]= (double)(~(int)(astack_d[atop-1]));
303          break;
304        case I_PRF | O_and : 
305          astack_d[atop-2]= (((int)(astack_d[atop-2])) & ((int)(astack_d[atop-1]))); atop--;
306          break;
307        case I_PRF | O_or : 
308          astack_d[atop-2]= (((int)(astack_d[atop-2])) | ((int)(astack_d[atop-1]))); atop--;
309          break;
310        case I_PRF | O_xor : 
311          astack_d[atop-2]= (((int)(astack_d[atop-2])) ^ ((int)(astack_d[atop-1]))); atop--;
312          break;
313
314        case I_PRF | T_Str | O_gt : 
315          astack_d[atop-2]= (astack_s[atop-2].compareTo(astack_s[atop-1])>0)?-1:0; atop--;
316          break;
317        case I_PRF | T_Str | O_lt : 
318          astack_d[atop-2]= (astack_s[atop-1].compareTo(astack_s[atop-2])>0)?-1:0; atop--;
319          break;
320        case I_PRF | T_Str | O_ne : 
321          astack_d[atop-2]= (astack_s[atop-2].equals(astack_s[atop-1]))?0:-1; atop--;
322          break;
323        case I_PRF | T_Str | O_eq : 
324          astack_d[atop-2]= (astack_s[atop-2].equals(astack_s[atop-1]))?-1:0; atop--;
325          break;
326        case I_PSH | T_Str | M_IMM : 
327          astack_s[atop++]=pargS[i];
328          break;
329        case I_PSH | T_Str | M_MEM : 
330          astack_s[atop++]=using_machine.variables.variablestring[pargmem[i]];
331          break;
332        case I_PSH | T_Str | M_MEMARR1 : 
333          astack_s[atop-1]=using_machine.variables.variablearraystring1[pargmem[i]][(int)astack_d[atop-1]];
334          break;
335        case I_PSH | T_Str | M_MEMARR2 : 
336          astack_s[atop-2]=using_machine.variables.variablearraystring2[pargmem[i]][(int)astack_d[atop-2]][(int)astack_d[atop-1]]; atop--;
337          break;
338        case I_STO | T_Str | M_MEM : 
339          using_machine.variables.variablestring[pargmem[i]]=astack_s[--atop];
340          break;
341        case I_STO | T_Str | M_MEMARR1 : 
342          using_machine.variables.variablearraystring1[pargmem[i]][(int)astack_d[atop-2]]=astack_s[atop-1];
343          atop--; atop--;
344          break;
345        case I_STO | T_Str | M_MEMARR2 : 
346          using_machine.variables.variablearraystring2[pargmem[i]][(int)astack_d[atop-3]][(int)astack_d[atop-2]]=astack_s[atop-1];
347          atop--; atop--; atop--;
348          break;
349
350        case I_PSH | T_Dbl | M_IMM : 
351          astack_d[atop++]=pargD[i];
352          break;
353        case I_PSH | T_Dbl | M_MEM : 
354          astack_d[atop++]=using_machine.variables.variablevalue[pargmem[i]];
355          break;
356        case I_PSH | T_Dbl | M_MEMARR1 : 
357          astack_d[atop-1]=using_machine.variables.variablearrayvalue1[pargmem[i]][(int)astack_d[atop-1]];
358          break;
359        case I_PSH | T_Dbl | M_MEMARR2 : 
360          astack_d[atop-2]=using_machine.variables.variablearrayvalue2[pargmem[i]][(int)astack_d[atop-2]][(int)astack_d[atop-1]];
361          atop--;
362          break;
363        case I_STO | T_Dbl | M_MEM : 
364          using_machine.variables.variablevalue[pargmem[i]]=astack_d[--atop];
365          break;
366        case I_STO | T_Dbl | M_MEMARR1 : 
367          using_machine.variables.variablearrayvalue1[pargmem[i]][(int)astack_d[atop-2]]=astack_d[atop-1];
368          atop--; atop--;
369          break;
370        case I_STO | T_Dbl | M_MEMARR2 : 
371          using_machine.variables.variablearrayvalue2[pargmem[i]][(int)astack_d[atop-3]][(int)astack_d[atop-2]]=astack_d[atop-1];
372          atop--; atop--; atop--;
373          break;
374        default:
375          if (verbose) System.out.printf("X ");
376          System.out.printf("Instruction Fault at %d instruction %d\n",i,prog[i]);
377          // should through an error! Instruction Fault
378          throw new EvaluateException("INSTRUCTION FAULT");              
379          //break;
380      }
381      if (verbose) System.out.printf("\n");
382    }
383          } catch(Exception e) {
384            //System.out.println(e.getMessage());
385            e.printStackTrace();
386            System.out.println(e);
387            throw new EvaluateException("BAD SUBSTRING INDEX");              
388          }
389    
390    //return nextpnt(x); // we've gone off the end
391    return -1;
392  }

427  void savestart(int x) { 
428          // if <0 then it is a rejected compile, so dont try to recompile, just nod and carry on...
429          if (acpointer[x]<0) return;
430          savestart_p=x; savestart_ac=top; tmptop=top; 
431          record=true; 
432          using_machine.evaluate_engine.speeder_compile=true;
433      // probably dont need both record and swith on speeder_compile
434  }
435  void saveelseacode(int then, int end) {
436    acpointer_next[then]=end;
437  }
438  void saveacode(int end) 
439  {
440          if (record) {
441            // push a HLT onto code
442            addInstr(I_HLT); 
443            acpointer[savestart_p]=savestart_ac;
444            acpointer_next[savestart_p]=end;
445            // this is now marked as a valid piece of code
446            top=tmptop;
447            if (using_machine.verbose) System.out.printf("Saved code from %d to %d for location %d\n",savestart_ac,top,savestart_p);
448            record=false;
449            using_machine.evaluate_engine.speeder_compile=false;
450          }
451  }

453  void reject()
454  {
455    acpointer[savestart_p]=-1; // don't try to run acode or recompile this
456    record=false;
457    using_machine.evaluate_engine.speeder_compile=false;
458  }
