634    /* create a circular string array */
635    /* you can PUSH onto the end or PULL from the end */
636    int maxsize=0;
637    int head=0;
638    int base=0;
639    int size=0;
640    char buffer[][];
641    void push(char charray[]) {
642      /* add this string to the array, wrapping and loosing oldest okay */      
643      size++;
644      if (verbose) System.out.printf("size = %d\n",size);
645      head=(head+1)%maxsize;
646      buffer[head]=charray;
647      /* check if overlapped */      
648      if (size>maxsize) { 
649        size--; 
650        base=(base+1)%maxsize;
651      }
652    }
653    char[] pull() {
654      int i;
655      if (size==0) return new char[0];
656      i=head;
657      size--;
658      head--; if (head<0) head=maxsize-1;
659      return buffer[i];
660    }
661    void flush() {
662      /* truncate it */
663      head=maxsize-1;
664      base=0;
665      size=0;
666    }
667    CircleStringArray(int maxsize) {
668      if (maxsize>0 && maxsize<100000) {
669        this.maxsize=maxsize;
670        buffer=new char[maxsize][];
671        head=maxsize-1;
672      }
673    }
